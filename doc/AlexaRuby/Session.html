def initialize(session)
  if session.nil? || session[:new].nil? || session[:sessionId].nil?
    raise ArgumentError, 'Invalid Session'
  end

  @new = session[:new]
  @session_id = session[:sessionId]
  @attributes = session[:attributes].nil? ? {} : session[:attributes]
  @user = session[:user]
end
