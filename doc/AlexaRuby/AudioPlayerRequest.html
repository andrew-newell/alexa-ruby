def initialize(json)
  @type = :audio_player
  req = json[:request]
  @playback_state = req[:type].gsub!('AudioPlayer.', '')
  @playback_offset = current_offset(req)
  define_error_params(req) if @playback_state == 'PlaybackFailed'
  super
end
